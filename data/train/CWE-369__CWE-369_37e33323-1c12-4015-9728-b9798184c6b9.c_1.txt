void CWE369_Divide_by_Zero__float_fscanf_18_bad()
{
    float data;
    /* Initialize data */
    data = 0.0F;
    goto source;
source:
    /* POTENTIAL FLAW: Use a value input from the console using fscanf() */
    fscanf (stdin, "%f", &data);
    goto sink;
sink:
    {
        /* POTENTIAL FLAW: Possibly divide by zero */
        int result = (int)(100.0 / data);
        printIntLine(result);
    }
}
