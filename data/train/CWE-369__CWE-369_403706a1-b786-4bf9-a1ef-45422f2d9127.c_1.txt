{
    float data;
    /* Initialize data */
    data = 0.0F;
    while(1)
    {
        /* POTENTIAL FLAW: Use a value input from the console using fscanf() */
        fscanf (stdin, "%f", &data);
        break;
    }
    while(1)
    {
        {

}
